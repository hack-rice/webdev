Javascript

Builds Functionality

Basic Javascript

function biggest (numbers) {
    var current = numbers[0];
    for (var i = 0; i < numbers.length; i++) {
        if (numbers[i] > current) {
            current = numbers[i];
        }
    };
    return current;
}
